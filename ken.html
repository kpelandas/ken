#include struct Contact { char name[50]; char phone[20]; char email[50]; char relationship[50]; }; void addContact(struct Contact contacts[], int *numContacts) { if (*numContacts < 100) { struct Contact newContact; printf("Enter contact name:"); scanf("%s", newContact.name); printf("Enter contact phone number for emergency: "); scanf("%s", newContact.phone); printf("Enter relationship status:"); scanf("%s", newContact.relationship); printf("Enter contact email: "); scanf("%s", newContact.email); contacts[*numContacts] = newContact; (*numContacts)++; printf("Contact added successfully!\n"); } else { printf("Contact list is full!\n"); } } void viewContacts(struct Contact contacts[], int numContacts) { if (numContacts > 0) { printf("\nList of Contacts:\n"); for (int i = 0; i < numContacts; ++i) { printf("Contact %d:\n", i + 1); printf("+-------------------------+\n"); printf("| Name: %-18s \n|", contacts[i].name); printf("| Phone: %-16s \n|", contacts[i].phone); printf("| Email: %-18s \n|", contacts[i].email); printf("| Relation: %-14s \n|", contacts[i].relationship); printf("+-------------------------+\n"); } } else { printf("No contacts to display!\n"); } } int main() { struct Contact contacts[100]; int numContacts = 0; int choice; do { printf("\nContact for Emergency\n\n"); printf("1. Add Contact\n"); printf("2. View Contacts\n"); printf("3. Exit\n"); printf("Enter your choice:"); scanf("%d", &choice); printf("\n"); switch (choice) { case 1: addContact(contacts, &numContacts); break; case 2: viewContacts(contacts, numContacts); break; case 3: printf("Exiting the program. Goodbye!\n"); exit(0); default: printf("Invalid choice. Please enter a valid option.\n"); } } while (choice != 3); return 0; }
